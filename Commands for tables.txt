CREATE TABLE users (
user_id INT(11) NOT NULL AUTO_INCREMENT,
username VARCHAR(255),
email VARCHAR(255),
created_at DATE,
updated_at DATE,
PRIMARY KEY(user_id)
);      
CREATE TABLE topics (
topic_id INT(11) NOT NULL AUTO_INCREMENT,
title VARCHAR(255),
post_body VARCHAR(2550),
user_id INT(11),
created_at DATE,
updated_at DATE,
PRIMARY KEY(topic_id)
);
CREATE TABLE replies (
reply_id INT(11) NOT NULL AUTO_INCREMENT,
reply_body VARCHAR(2550),
user_id INT(11),
topic_id INT (11),
created_at DATE,
updated_at DATE,
PRIMARY KEY(reply_id)
);
CREATE TABLE watchers (
watcher_id INT(11) NOT NULL AUTO_INCREMENT,
user_id INT(11),
topic_id INT(11),
PRIMARY KEY(watcher_id)
);
